    ```dart
    import 'package:flutter/material.dart';
    import 'package:flutter_riverpod/flutter_riverpod.dart';
    import '../../app/theme.dart';
    import '../../shared/repositories/workout_repository.dart';

    // --- Static Content Constants ---

    const _kMobilityItems = [
      _RestItem('Hip Flexor Stretch', '60 sec each side • releases tight hip flexors'),
      _RestItem('Thoracic Rotation', '10 reps each side • improves upper back mobility'),
      _RestItem('Shoulder Circles', '15 reps forward + back • warms up shoulder joint'),
      _RestItem('Cat-Cow', '10 slow reps • spinal decompression'),
      _RestItem('90/90 Hip Stretch', '90 sec each side • external rotation'),
    ];

    const _kFoamRollingItems = [
      _RestItem('Quads', '60–90 sec • roll slowly, pause on tight spots'),
      _RestItem('Upper Back', '60 sec • cross arms, roll T-spine'),
      _RestItem('IT Band', '45 sec each side • side-lying, roll outer thigh'),
      _RestItem('Calves', '60 sec each • point and flex foot while rolling'),
      _RestItem('Lats', '45 sec each side • arm overhead, roll side of back'),
    ];

    const _kActiveRecoveryItems = [
      _RestItem('Light Walk', '20–30 min • low heart rate, fresh air'),
      _RestItem('Swimming', '20 min easy • full-body low-impact recovery'),
      _RestItem('Cycling', '15–20 min easy pace • promotes blood flow'),
    ];

    class _RestItem {
      final String name;
      final String detail;
      const _RestItem(this.name, this.detail);
    }

    /// Displays recovery content for rest days.
    ///
    /// Contains 4 sections: Mobility, Foam Rolling, Active Recovery, This Week.
    /// This Week reads from WorkoutRepository to show real weekly stats.
    /// Safe to render with zero workout history (first-launch guard).
    class RestDayContent extends ConsumerWidget {
      const RestDayContent({super.key});

      @override
      Widget build(BuildContext context, WidgetRef ref) {
        // Weekly stats (safe defaults when no history)
        final weekStart = DateTime.now().subtract(Duration(days: DateTime.now().weekday - 1));
        final weekWorkouts = WorkoutRepository().getForWeek(weekStart);
        final workoutCount = weekWorkouts.length;

        return Column(
          crossAxisAlignment: CrossAxisAlignment.start,
          children: [
            _sectionHeader('TODAY\'S RECOVERY'),
            const SizedBox(height: AppSpacing.sm),
            _recoverySection(
              icon: Icons.self_improvement,
              title: 'MOBILITY',
              color: AppColors.accent,
              items: _kMobilityItems,
            ),
            const SizedBox(height: AppSpacing.md),
            _recoverySection(
              icon: Icons.compress,
              title: 'FOAM ROLLING',
              color: AppColors.sky,
              items: _kFoamRollingItems,
            ),
            const SizedBox(height: AppSpacing.md),
            _recoverySection(
              icon: Icons.directions_walk,
              title: 'ACTIVE RECOVERY',
              color: AppColors.warm,
              items: _kActiveRecoveryItems,
            ),
            const SizedBox(height: AppSpacing.lg),
            _weeklyReflection(workoutCount),
            const SizedBox(height: AppSpacing.xxl),
          ],
        );
      }

      Widget _sectionHeader(String label) {
        return Text(
          label,
          style: AppTextStyles.labelUppercase(AppColors.textSecondary),
        );
      }

      Widget _recoverySection({
        required IconData icon,
        required String title,
        required Color color,
        required List<_RestItem> items,
      }) {
        return Container(
          padding: const EdgeInsets.all(AppSpacing.lg),
          decoration: BoxDecoration(
            color: AppColors.bgCard,
            borderRadius: BorderRadius.circular(AppRadius.lg),
            border: Border.all(color: AppColors.border, width: 0.5),
          ),
          child: Column(
            crossAxisAlignment: CrossAxisAlignment.start,
            children: [
              Row(
                children: [
                  Icon(icon, color: color, size: 16),
                  const SizedBox(width: AppSpacing.sm),
                  Text(title, style: AppTextStyles.labelUppercase(color)),
                ],
              ),
              const SizedBox(height: AppSpacing.md),
              ...items.map((item) => _restItemRow(item)),
            ],
          ),
        );
      }

      Widget _restItemRow(_RestItem item) {
        return Padding(
          padding: const EdgeInsets.only(bottom: AppSpacing.sm),
          child: Row(
            crossAxisAlignment: CrossAxisAlignment.start,
            children: [
              Container(
                width: 4,
                height: 4,
                margin: const EdgeInsets.only(top: 7),
                decoration: const BoxDecoration(
                  color: AppColors.textTertiary,
                  shape: BoxShape.circle,
                ),
              ),
              const SizedBox(width: AppSpacing.md),
              Expanded(
                child: Column(
                  crossAxisAlignment: CrossAxisAlignment.start,
                  children: [
                    Text(item.name,
                        style: AppTextStyles.bodyStrong(AppColors.textPrimary)),
                    Text(item.detail,
                        style: AppTextStyles.caption(AppColors.textSecondary)),
                  ],
                ),
              ),
            ],
          ),
        );
      }

      Widget _weeklyReflection(int workoutCount) {
        final sessionWord = workoutCount == 1 ? 'session' : 'sessions';
        final reflectionText = workoutCount == 0
            ? 'No sessions logged yet this week. Rest days count too.'
            : 'You\'ve completed $workoutCount $sessionWord this week. '
                'Rest is part of the plan — your body adapts while you recover.';

        return Container(
          padding: const EdgeInsets.all(AppSpacing.lg),
          decoration: BoxDecoration(
            color: AppColors.bgCard,
            borderRadius: BorderRadius.circular(AppRadius.lg),
            border: Border.all(color: AppColors.border, width: 0.5),
          ),
          child: Column(
            crossAxisAlignment: CrossAxisAlignment.start,
            children: [
              Row(
                children: [
                  const Icon(Icons.bar_chart, color: AppColors.warm, size: 16),
                  const SizedBox(width: AppSpacing.sm),
                  Text('THIS WEEK',
                      style: AppTextStyles.labelUppercase(AppColors.warm)),
                ],
              ),
              const SizedBox(height: AppSpacing.md),
              Text(reflectionText,
                  style: AppTextStyles.body(AppColors.textSecondary)),
              if (workoutCount > 0) ...[
                const SizedBox(height: AppSpacing.sm),
                Text(
                  '$workoutCount × training',
                  style: AppTextStyles.bodyStrong(AppColors.textPrimary),
                ),
              ],
            ],
          ),
        );
      }
    }
    ```
