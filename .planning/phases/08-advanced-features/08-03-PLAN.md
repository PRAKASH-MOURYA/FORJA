pw.Widget _buildVolumeBars(List<double> weeklyVolumes) {
  final maxVol = weeklyVolumes.isEmpty ? 1.0 : weeklyVolumes.reduce(max);
  return pw.Row(

    **Step 2 — ExportService:** Create `lib/shared/services/export_service.dart`. Accept repositories via constructor for testability. Use `getTemporaryDirectory()` + `SharePlus.instance.shareXFiles()` for delivery (NOT getDownloadsDirectory).

    ```dart
    import 'dart:io';
    import 'dart:math';
    import 'package:path_provider/path_provider.dart';
    import 'package:pdf/pdf.dart';
    import 'package:pdf/widgets.dart' as pw;
    import 'package:share_plus/share_plus.dart';
    import '../repositories/workout_repository.dart';
    import '../repositories/pr_repository.dart';

    class ExportService {
      final WorkoutRepository _workoutRepo;
      final PRRepository _prRepo;

      ExportService({
        required WorkoutRepository workoutRepo,
        required PRRepository prRepo,
      })  : _workoutRepo = workoutRepo,
            _prRepo = prRepo;

      // --- CSV ---

      Future<String> buildCsvString() async {
        final buf = StringBuffer();
        buf.writeln('date,workout_id,exercise_name,set_number,weight_kg,reps,completed,is_pr');
        final workouts = _workoutRepo.getAll();
        for (final w in workouts) {
          for (final s in w.sets) {
            final isPr = _prRepo.isPersonalRecord(s.exerciseId, s.weightKg, s.reps);
            buf.writeln([
              w.startedAt.toIso8601String(),
              w.id,
              s.exerciseId,
              s.setNumber,
              s.weightKg,
              s.reps,
              s.completed,
              isPr,
            ].join(','));
          }
        }
        return buf.toString();
      }

      Future<void> exportCsv() async {
        final csv = await buildCsvString();
        final dir = await getTemporaryDirectory();
        final now = DateTime.now();
        final filename = 'forja_export_${now.year}-${now.month.toString().padLeft(2, '0')}.csv';
        final file = File('${dir.path}/$filename');
        await file.writeAsString(csv);
        await SharePlus.instance.shareXFiles([XFile(file.path)], subject: filename);
      }

      // --- PDF ---

      Future<pw.Document> buildPdf() async {
        final workouts = _workoutRepo.getAll();
        final prs = _prRepo.getAll();
        final now = DateTime.now();
        final monthLabel = '${now.year}-${now.month.toString().padLeft(2, '0')}';
        final isLimited = workouts.length < 4;

        final doc = pw.Document();
        doc.addPage(pw.MultiPage(
          pageFormat: PdfPageFormat.a4,
          margin: const pw.EdgeInsets.all(32),
          header: (_) => pw.Text(
            isLimited ? 'Your progress so far' : 'Monthly Progress — $monthLabel',
            style: pw.TextStyle(fontSize: 20, fontWeight: pw.FontWeight.bold),
          ),
          build: (ctx) => [
            pw.SizedBox(height: 16),
            pw.Text('Volume by Week', style: pw.TextStyle(fontSize: 14, fontWeight: pw.FontWeight.bold)),
            pw.SizedBox(height: 8),
            _buildVolumeBars(_weeklyVolumes(workouts)),
            pw.SizedBox(height: 24),
            pw.Text('Personal Records', style: pw.TextStyle(fontSize: 14, fontWeight: pw.FontWeight.bold)),
            pw.SizedBox(height: 8),
            _buildPrTable(prs),
            pw.SizedBox(height: 24),
            pw.Text('Workout Consistency', style: pw.TextStyle(fontSize: 14, fontWeight: pw.FontWeight.bold)),
            pw.SizedBox(height: 8),
            _buildConsistencyTable(workouts),
          ],
        ));
        return doc;
      }

      Future<void> exportPdf() async {
        final doc = await buildPdf();
        final bytes = await doc.save();
        final dir = await getTemporaryDirectory();
        final now = DateTime.now();
        final filename = 'forja_progress_${now.year}-${now.month.toString().padLeft(2, '0')}.pdf';
        final file = File('${dir.path}/$filename');
        await file.writeAsBytes(bytes);
        await SharePlus.instance.shareXFiles([XFile(file.path)], subject: filename);
      }

      // --- Helpers ---

      List<double> _weeklyVolumes(List workouts) {
        // Group workouts by ISO week number; sum volume per week (last 4 weeks)
        final Map<int, double> weekVolumes = {};
        for (final w in workouts) {
          final week = _isoWeek(w.startedAt);
          final vol = (w.sets as List).where((s) => s.completed).fold<double>(
            0, (sum, s) => sum + s.weightKg * s.reps);
          weekVolumes[week] = (weekVolumes[week] ?? 0) + vol;
        }
        final sorted = weekVolumes.entries.toList()..sort((a, b) => a.key.compareTo(b.key));
        return sorted.map((e) => e.value).toList();
      }

      int _isoWeek(DateTime date) {
        final startOfYear = DateTime(date.year, 1, 1);
        return ((date.difference(startOfYear).inDays) / 7).floor() + 1;
      }

      pw.Widget _buildVolumeBars(List<double> weeklyVolumes) {
        if (weeklyVolumes.isEmpty) {
          return pw.Text('No workout data yet.', style: pw.TextStyle(color: PdfColors.grey));
        }
        final maxVol = weeklyVolumes.reduce(max);
        return pw.Row(
          crossAxisAlignment: pw.CrossAxisAlignment.end,
          children: weeklyVolumes.asMap().entries.map((e) {
            final height = maxVol > 0 ? (e.value / maxVol) * 80 : 4.0;
            return pw.Padding(
              padding: const pw.EdgeInsets.symmetric(horizontal: 4),
              child: pw.Column(children: [
                pw.Container(width: 24, height: height, color: PdfColors.blue700),
                pw.Text('W${e.key + 1}', style: pw.TextStyle(fontSize: 8)),
              ]),
            );
          }).toList(),
        );
      }

      pw.Widget _buildPrTable(List prs) {
        if (prs.isEmpty) {
          return pw.Text('No personal records yet.', style: pw.TextStyle(color: PdfColors.grey));
        }
        return pw.Table.fromTextArray(
          border: pw.TableBorder.all(),
          headerStyle: pw.TextStyle(fontWeight: pw.FontWeight.bold, fontSize: 10),
          cellStyle: const pw.TextStyle(fontSize: 9),
          headers: ['Exercise', 'Weight (kg)', 'Reps', 'Est. 1RM', 'Date'],
          data: prs.map((pr) => [
            pr.exerciseId ?? '',
            pr.weightKg?.toString() ?? '',
            pr.reps?.toString() ?? '',
            pr.estimated1RM?.toStringAsFixed(1) ?? '',
            pr.achievedAt?.toIso8601String().split('T').first ?? '',
          ]).toList(),
        );
      }

      pw.Widget _buildConsistencyTable(List workouts) {
        if (workouts.isEmpty) {
          return pw.Text('No workouts logged yet.', style: pw.TextStyle(color: PdfColors.grey));
        }
        // Count workouts per ISO week
        final Map<int, int> weekCounts = {};
        for (final w in workouts) {
          final week = _isoWeek(w.startedAt);
          weekCounts[week] = (weekCounts[week] ?? 0) + 1;
        }
        final sorted = weekCounts.entries.toList()..sort((a, b) => a.key.compareTo(b.key));
        return pw.Table.fromTextArray(
          border: pw.TableBorder.all(),
          headerStyle: pw.TextStyle(fontWeight: pw.FontWeight.bold, fontSize: 10),
          cellStyle: const pw.TextStyle(fontSize: 9),
          headers: ['Week', 'Workouts'],
          data: sorted.map((e) => ['Week ${e.key}', e.value.toString()]).toList(),
        );
      }
    }
    ```

    import 'package:flutter/material.dart';
    import 'package:hooks_riverpod/hooks_riverpod.dart';
    import 'package:flutter_hooks/flutter_hooks.dart';
    import '../../../shared/services/export_service.dart';
    // import WorkoutRepository + PRRepository providers

    class ExportSection extends HookConsumerWidget {
      const ExportSection({super.key});

      @override
      Widget build(BuildContext context, WidgetRef ref) {
        final csvLoading = useState(false);
        final pdfLoading = useState(false);

        Future<void> handleExportCsv() async {
          csvLoading.value = true;
          try {
            final workoutRepo = ref.read(workoutRepositoryProvider);
            final prRepo = ref.read(prRepositoryProvider);
            final svc = ExportService(workoutRepo: workoutRepo, prRepo: prRepo);
            await svc.exportCsv();
          } catch (e) {
            if (context.mounted) {
              ScaffoldMessenger.of(context).showSnackBar(
                SnackBar(content: Text('Export failed: ${e.toString()}')),
              );
            }
          } finally {
            csvLoading.value = false;
          }
        }

        Future<void> handleExportPdf() async {
          pdfLoading.value = true;
          try {
            final workoutRepo = ref.read(workoutRepositoryProvider);
            final prRepo = ref.read(prRepositoryProvider);
            final svc = ExportService(workoutRepo: workoutRepo, prRepo: prRepo);
            await svc.exportPdf();
          } catch (e) {
            if (context.mounted) {
              ScaffoldMessenger.of(context).showSnackBar(
                SnackBar(content: Text('Export failed: ${e.toString()}')),
              );
            }
          } finally {
            pdfLoading.value = false;
          }
        }

        return Column(
          crossAxisAlignment: CrossAxisAlignment.start,
          children: [
            const Padding(
              padding: EdgeInsets.symmetric(horizontal: 16, vertical: 8),
              child: Text('Export Data', style: TextStyle(fontSize: 16, fontWeight: FontWeight.bold)),
            ),
            ListTile(
              leading: const Icon(Icons.table_chart_outlined),
              title: const Text('Export CSV'),
              subtitle: const Text('Full workout history, one row per set'),
              trailing: csvLoading.value
                  ? const SizedBox(width: 20, height: 20, child: CircularProgressIndicator(strokeWidth: 2))
                  : const Icon(Icons.share),
              onTap: csvLoading.value ? null : handleExportCsv,
            ),
            ListTile(
              leading: const Icon(Icons.picture_as_pdf_outlined),
              title: const Text('Export PDF'),
              subtitle: const Text('Monthly progress summary with charts'),
              trailing: pdfLoading.value
                  ? const SizedBox(width: 20, height: 20, child: CircularProgressIndicator(strokeWidth: 2))
                  : const Icon(Icons.share),
              onTap: pdfLoading.value ? null : handleExportPdf,
            ),
          ],
        );
      }
    }
