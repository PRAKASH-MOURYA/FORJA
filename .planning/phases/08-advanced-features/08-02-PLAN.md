    **Step 2 — PlateVisualCard:** Create `lib/features/today/widgets/plate_visual_card.dart` using CustomPainter for the plate illustration. Two variants — training day (higher carbs) and rest day (higher protein/fat, fewer carbs).

    Plate is drawn as a circle divided into sections using CustomPainter:
    - Training day sections: Carbs 45% (amber), Protein 35% (orange), Fat 20% (blueGrey)

    ```dart
    import 'dart:math';
    import 'package:flutter/material.dart';
    import 'package:hooks_riverpod/hooks_riverpod.dart';
    // import adaptiveTodayProvider

    class PlateVisualCard extends ConsumerWidget {
      const PlateVisualCard({super.key});

      @override
      Widget build(BuildContext context, WidgetRef ref) {
        final today = ref.watch(adaptiveTodayProvider).valueOrNull;
        final isRestDay = today?.isRestDay ?? false;

        return Card(
          child: Padding(
            padding: const EdgeInsets.all(16),
            child: Column(
              crossAxisAlignment: CrossAxisAlignment.start,
              children: [
                Text(isRestDay ? 'Rest Day Plate' : 'Training Day Plate',
                    style: const TextStyle(fontWeight: FontWeight.bold)),
                const SizedBox(height: 8),
                Row(children: [
                  SizedBox(
                    width: 80, height: 80,
                    child: CustomPaint(painter: _PlatePainter(isRestDay: isRestDay)),
                  ),
                  const SizedBox(width: 16),
                  _buildLegend(isRestDay),
                ]),
              ],
            ),
          ),
        );
      }

      Widget _buildLegend(bool isRestDay) {
        final items = isRestDay
            ? [('Protein', Colors.orange, '45%'), ('Fat', Colors.blueGrey, '30%'), ('Carbs', Colors.amber, '25%')]
            : [('Carbs', Colors.amber, '45%'), ('Protein', Colors.orange, '35%'), ('Fat', Colors.blueGrey, '20%')];
        return Column(
          crossAxisAlignment: CrossAxisAlignment.start,
          children: items.map((item) => Padding(
            padding: const EdgeInsets.symmetric(vertical: 2),
            child: Row(children: [
              Container(width: 12, height: 12, decoration: BoxDecoration(color: item.$2, shape: BoxShape.circle)),
              const SizedBox(width: 6),
              Text('${item.$1} ${item.$3}', style: const TextStyle(fontSize: 12)),
            ]),
          )).toList(),
        );
      }
    }

    class _PlatePainter extends CustomPainter {
      final bool isRestDay;
      _PlatePainter({required this.isRestDay});

      @override
      void paint(Canvas canvas, Size size) {
        final sections = isRestDay
            ? [(0.45, Colors.orange), (0.30, Colors.blueGrey), (0.25, Colors.amber)]
            : [(0.45, Colors.amber), (0.35, Colors.orange), (0.20, Colors.blueGrey)];
        final center = Offset(size.width / 2, size.height / 2);
        final radius = min(size.width, size.height) / 2;
        var startAngle = -pi / 2;
        for (final s in sections) {
          final sweepAngle = s.$1 * 2 * pi;
          canvas.drawArc(
            Rect.fromCircle(center: center, radius: radius),
            startAngle, sweepAngle, true,
            Paint()..color = s.$2,
          );
          startAngle += sweepAngle;
        }
        // White border
        canvas.drawCircle(center, radius, Paint()..color = Colors.white..style = PaintingStyle.stroke..strokeWidth = 2);
      }

      @override
      bool shouldRepaint(_PlatePainter old) => old.isRestDay != isRestDay;
    }
    ```
