    ```dart
    class StreakService {
      // evaluateStreak: pure function, computes ISO week numbers using _isoWeek()
      // Same week → return profile unchanged
      // Gap == 1 week → copyWith(streakWeeks: streakWeeks + 1)
      // Gap > 1, shields > 0 → copyWith(streakShields: shields - 1)
      // Gap > 1, shields == 0 → copyWith(streakWeeks: 0)
      static int _isoWeek(DateTime date) {
        final jan4 = DateTime(date.year, 1, 4);
        return ((date.difference(jan4).inDays + jan4.weekday) / 7).ceil();
      }
    }
    ```
