```dart
// Muscle group canonical names (6 groups):
// 'Chest', 'Back', 'Legs', 'Shoulders', 'Arms', 'Core'

enum MuscleRecoveryZone { green, yellow, red }

class MuscleRecoveryStatus {
  final String muscle;
  final MuscleRecoveryZone zone;
  const MuscleRecoveryStatus({required this.muscle, required this.zone});
}

class MuscleRecoveryService {
  // Map exercise.muscle → canonical group
  static const _muscleMap = {
    'Chest': 'Chest', 'Upper Chest': 'Chest', 'Lower Chest': 'Chest',
    'Back': 'Back', 'Lats': 'Back', 'Upper Back': 'Back', 'Mid Back': 'Back',
    'Traps': 'Back',
    'Legs': 'Legs', 'Quads': 'Legs', 'Hamstrings': 'Legs', 'Glutes': 'Legs',
    'Calves': 'Legs',
    'Shoulders': 'Shoulders', 'Front Delts': 'Shoulders',
    'Side Delts': 'Shoulders', 'Rear Delts': 'Shoulders',
    'Arms': 'Arms', 'Biceps': 'Arms', 'Triceps': 'Arms', 'Forearms': 'Arms',
    'Core': 'Core', 'Abs': 'Core', 'Obliques': 'Core',
  };

  static const _canonicalGroups = ['Chest', 'Back', 'Legs', 'Shoulders', 'Arms', 'Core'];

  List<MuscleRecoveryStatus> getRecoveryStatuses() {
    // Use WorkoutRepository + CheckInRepository — instantiate internally
    // For each canonical group:
    //   1. Find the most recent WorkoutLog date that included this muscle group
    //      (scan all SetLogs, filter by exerciseId matching exercises with that muscle group)
    //      Since SetLogs store exerciseId not muscle, use workoutLog.startedAt as proxy:
    //      scan all setLogs for completed sets, map exerciseId to muscle via kExerciseData,
    //      find most recent createdAt for sets in this muscle group
    //   2. daysSinceLastTrained = DateTime.now().difference(lastTrainedDate).inDays (or 99 if never)
    //   3. Get latest CheckIn soreness (1-5), default 1 if no check-in
    //   4. Zone logic:
    //      if daysSince >= 3 && soreness <= 2 → green
    //      if daysSince >= 2 && soreness <= 3 → green
    //      if (daysSince <= 1) || soreness >= 4 → red
    //      else → yellow
    // Return list of 6 MuscleRecoveryStatus in _canonicalGroups order
  }

  String summaryText(List<MuscleRecoveryStatus> statuses) {
    final ready = statuses.where((s) => s.zone == MuscleRecoveryZone.green).length;
    final fatigued = statuses.length - ready;
    if (fatigued == 0) return 'All muscles ready';
    return '$ready ready · $fatigued fatigued';
  }
}
```

Zone logic (precise):
- `daysSince >= 3 && soreness <= 3` → green
- `daysSince >= 2 && soreness <= 2` → green
- `daysSince == 0 || soreness >= 4` → red
- everything else → yellow

```dart
// --- pre-workout data (synchronous Hive reads) ---
final workoutRepo = WorkoutRepository();
final prRepo = PrRepository();
final allLogs = workoutRepo.getAll();

// Build lastSession map: exerciseId → "Last: Xkg × Y"
final Map<String, String> lastSessionMap = {};
for (final exercise in sessionExercises) {
  for (final log in allLogs) {
    final sets = workoutRepo.getSetsForWorkout(log.id)
        .where((s) => s.exerciseId == exercise.id && s.completed)
        .toList();
    if (sets.isNotEmpty) {
      // Use the set with highest weight as representative
      final best = sets.reduce((a, b) => a.weightKg > b.weightKg ? a : b);
      lastSessionMap[exercise.id] = 'Last: ${best.weightKg.toStringAsFixed(0)}kg × ${best.reps}';
      break; // most recent log found — stop
    }
  }
}

// PR to Beat: find best PR opportunity across today's exercises
String? prExerciseName;
double prCurrentKg = 0;
double prTargetKg = 0;
for (final exercise in sessionExercises) {
  final pr = prRepo.getLatestPRForExercise(exercise.id);
  if (pr != null) {
    final weightKg = (pr['weight_kg'] as num).toDouble();
    if (prExerciseName == null || weightKg > prCurrentKg) {
      prExerciseName = exercise.name;
      prCurrentKg = weightKg;
      prTargetKg = prCurrentKg + 2.5;
    }
  }
}

// Recovery heatmap
final recoveryService = MuscleRecoveryService();
final recoveryStatuses = recoveryService.getRecoveryStatuses();
final recoverySummary = recoveryService.summaryText(recoveryStatuses);
```
