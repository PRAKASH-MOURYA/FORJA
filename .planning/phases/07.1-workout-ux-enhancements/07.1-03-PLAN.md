```dart
import 'package:freezed_annotation/freezed_annotation.dart';
import 'package:hive_flutter/hive_flutter.dart';

part 'custom_split.freezed.dart';
part 'custom_split.g.dart';

@freezed
@HiveType(typeId: 6)
class CustomSplit with _$CustomSplit {
  const factory CustomSplit({
    @HiveField(0) required String id,          // UUID string
    @HiveField(1) required String name,         // e.g. "My Push/Pull/Legs"
    @HiveField(2) required int daysCount,       // 2-6
    @HiveField(3) required List<SplitDay> days, // one per day
    @HiveField(4) required DateTime createdAt,
  }) = _CustomSplit;

  factory CustomSplit.fromJson(Map<String, dynamic> json) =>
      _$CustomSplitFromJson(json);
}

@freezed
@HiveType(typeId: 7)   // SplitDay needs its own typeId
class SplitDay with _$SplitDay {
  const factory SplitDay({
    @HiveField(0) required String dayName,          // e.g. "Push A"
    @HiveField(1) required List<String> exerciseIds, // from kExerciseData keys
  }) = _SplitDay;

  factory SplitDay.fromJson(Map<String, dynamic> json) =>
      _$SplitDayFromJson(json);
}
```

**Save logic (`_saveSplit`) — writes to both HiveService and UserProfile:**
```dart
Future<void> _saveSplit() async {
  final name = _splitNameController.text.trim();
  if (name.isEmpty) {
    ScaffoldMessenger.of(context).showSnackBar(
      const SnackBar(content: Text('Please name your split.')));
    return;
  }
  final split = CustomSplit(
    id: DateTime.now().millisecondsSinceEpoch.toString(),
    name: name,
    daysCount: _daysCount,
    days: _days.map((d) => SplitDay(
      dayName: d.dayName,
      exerciseIds: List.from(d.exerciseIds),
    )).toList(),
    createdAt: DateTime.now(),
  );
  // Persist the split itself
  await HiveService.customSplits.put(split.id, split);
  // Update UserProfile so ProfileScreen label and other consumers know the active split
  final profile = ref.read(userProfileProvider);
  if (profile != null) {
    await ref.read(userProfileProvider.notifier).update(
      (p) => p.copyWith(customSplitId: split.id),
    );
  }
  if (mounted) context.pop();
}
```

Note: `userProfileProvider.notifier.update` follows the existing pattern used elsewhere in the app for profile mutations. If the notifier exposes a different mutation method (e.g., `save`, `updateProfile`), use that instead — the intent is to write `profile.copyWith(customSplitId: split.id)` back through the same Riverpod + Hive path that all other profile fields use.
