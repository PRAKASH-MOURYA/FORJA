// White card with shadow, no border, configurable radius
class FCard extends StatelessWidget {
  final Widget child;
  final EdgeInsetsGeometry? padding;
  final double radius;        // default 18
  final List<BoxShadow>? shadows;
  const FCard({super.key, required this.child, this.padding, this.radius = 18, this.shadows});
  @override
  Widget build(BuildContext context) {
    return Container(
      padding: padding ?? const EdgeInsets.all(16),
      decoration: BoxDecoration(
        color: AppColors.bgCardNew,
        borderRadius: BorderRadius.circular(radius),
        boxShadow: shadows ?? [
          BoxShadow(
            color: const Color(0x1A6496BE),
            blurRadius: 12,
            offset: const Offset(0, 2),
          ),
        ],
      ),
      child: child,
    );
  }
}
